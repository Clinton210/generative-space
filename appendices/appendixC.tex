\chapter{Technical Lemmas for the Meta-Diagonalizer}

\section{Introduction}

This appendix contains the technical lemmas used in Chapter~8 to construct the
meta-diagonalizer.  
The results describe how computable projections depend on finite prefixes,
how tail modifications preserve projection values below the dependency horizon,
and how controlled adjustments can force divergence from any given family of
observers.

All notation and assumptions match the conventions of Chapters~6--9 and the
dependency bound framework developed in Appendix~B.

\section{Prefix Stabilization}

Computable projections depend only on a finite prefix of their input.  
The following lemma restates this principle for single projections.

\begin{lemma}[Prefix Stabilization]
\label{lem:prefix-stabilization}
Let $\Phi : \mathcal{G}_{\mathrm{eff}} \to \mathbb{R}$ be a computable
secondary projection with dependency bound $B_{\Phi}(\varepsilon)$.  
If effective identities $G$ and $H$ satisfy
\[
(M_G,D_G,K_G){\upharpoonright}B_{\Phi}(\varepsilon)
=
(M_H,D_H,K_H){\upharpoonright}B_{\Phi}(\varepsilon),
\]
then
\[
|\Phi(G) - \Phi(H)| < \varepsilon.
\]
\end{lemma}

\begin{proof}
A computable approximation to $\Phi$ reads only finitely many input symbols
before producing an $\varepsilon$-accurate output.  
If the inputs agree on this prefix, the resulting approximations agree within
the desired bound.
\end{proof}

\section{Uniform Stabilization for Finite Families}

Finite families of projections admit a single stabilization horizon.

\begin{lemma}[Uniform Stabilization]
\label{lem:uniform-stabilization}
Let $\mathcal{F} = \{\Phi_1,\ldots,\Phi_m\}$ be a finite family of computable
secondary projections, and let $B(\varepsilon)$ be the uniform dependency bound
of Proposition~\ref{prop:uniform-bound}.  
If $G$ and $H$ agree on the first $B(\varepsilon)$ coordinates, then
\[
|\Phi_i(G) - \Phi_i(H)| < \varepsilon
\qquad\text{for every } i=1,\ldots,m.
\]
\end{lemma}

\begin{proof}
Since
\[
B(\varepsilon)=\max_{i} B_{\Phi_i}(\varepsilon),
\]
agreement on the first $B(\varepsilon)$ coordinates yields agreement on each
prefix required for stabilization of each $\Phi_i$.  
Apply Lemma~\ref{lem:prefix-stabilization}.
\end{proof}

\section{Tail Sewing}

The meta-diagonalizer modifies the tail of a generator without altering what is
visible to a finite-prefix observer.  
The next lemma formalizes this.

\begin{lemma}[Tail Sewing]
\label{lem:tail-sewing}
Let $G$ and $H$ be effective identities, and let $L \in \mathbb{N}$.  
Define a new identity $G'$ by
\[
G'(n)=
\begin{cases}
G(n), & n \le L,\\
H(n), & n > L.
\end{cases}
\]
If $L \ge B(\varepsilon)$, where $B$ is the uniform dependency bound for a
finite family $\mathcal{F}$, then for each $\Phi_i \in \mathcal{F}$,
\[
|\Phi_i(G') - \Phi_i(G)| < \varepsilon.
\]
\end{lemma}

\begin{proof}
The identities $G'$ and $G$ agree on the first $L \ge B(\varepsilon)$
coordinates.
